Moon Slider
An open source, lightweight (4kb) JavaScript plugin for circular inputs.
Works on modern desktop browsers (Chrome, Firefox, Safari).

Also supports Android and iOS devices, but needs to be tested on more devices - so far tested on Android (Galaxy S3, 4.3), iPhone 5, with acceptable results.
Download on Github
Usage:
var my_slider = generateMoonSlider(

// submit the id of the div that will hold the slider
"your-div-id",

// optional - an object containing the option values
{
// the minimum value
min_value: 0,

// the maximum value
max_value: 1000,

// the radius of the circle
radius: 167,

// you can define the increment - default is 1
step: 10,

// color of the line
color: "#7a4b8e",

// start value - default is min_value
start_value: 750,

// slider css classname
slider_classname: "moon-slider",

// button css classname
button_classname: "moon-slider-button",

// color of the underlying (grey) line
base_line_color: "#d8d8d8"

},

/* optional - an object containing event handlers
(currently only the move handler is supported) */
{

onMoveHandler: function(event) {
/* the event object returned contains:
id, (the id of corresponding the div)
deg, (current degrees of the slider)
value (current value)
*/
}
}
);


/*
*
* API
*
*/

// set the value - the slider will automatically adjust
my_slider.setValue(100);


/* set the degrees
the slider value will be calculated automatically */
my_slider.setDeg(90);


// get the current value
var value = my_slider.getValue();


// get the current degrees
var degrees = my_slider.getDeg();
